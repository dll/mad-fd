// 微服务架构教学
[ApiController]
[Route("api/[controller]")]
public class LearningController : ControllerBase
{
    private readonly ILearningService _learningService;
    private readonly INotificationService _notificationService;
    
    public LearningController(
        ILearningService learningService,
        INotificationService notificationService)
    {
        _learningService = learningService;
        _notificationService = notificationService;
    }
    
    [HttpGet("courses")]
    public async Task<ActionResult<List<Course>>> GetCourses(
        [FromQuery] CourseQuery query)
    {
        var courses = await _learningService.GetCoursesAsync(query);
        return Ok(courses);
    }
    
    [HttpPost("enroll")]
    public async Task<ActionResult> EnrollCourse(EnrollRequest request)
    {
        var result = await _learningService.EnrollCourseAsync(request);
        
        if (result.Success)
        {
            // 发送通知
            await _notificationService.SendEnrollmentNotificationAsync(
                request.UserId, request.CourseId);
        }
        
        return Ok(result);
    }
}
