## 🔧 软件环境配置

### 开发工具统一配置
```bash
# Windows开发环境自动化配置脚本
# install_dev_env.ps1

# 1. 安装Chocolatey包管理器
Set-ExecutionPolicy Bypass -Scope Process -Force
iex ((New-Object System.Net.WebClient).DownloadString('https://chocolatey.org/install.ps1'))

# 2. 安装基础开发工具
choco install -y git
choco install -y nodejs --version=18.17.0
choco install -y python --version=3.11.0
choco install -y openjdk11
choco install -y vscode
choco install -y androidstudio

# 3. 配置Android SDK
$env:ANDROID_HOME = "C:\Android\android-sdk"
$env:PATH += ";$env:ANDROID_HOME\tools;$env:ANDROID_HOME\platform-tools"

# 4. 安装Flutter
git clone https://github.com/flutter/flutter.git -b stable C:\flutter
$env:PATH += ";C:\flutter\bin"

# 5. 验证安装
flutter doctor
git --version
node --version
java -version
```

```bash
# macOS开发环境配置脚本
# install_dev_env.sh

#!/bin/bash

# 1. 安装Homebrew
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# 2. 安装基础工具
brew install git
brew install node@18
brew install python@3.11
brew install openjdk@11
brew install --cask visual-studio-code
brew install --cask android-studio

# 3. 安装Xcode Command Line Tools
sudo xcode-select --install

# 4. 安装Flutter
git clone https://github.com/flutter/flutter.git -b stable ~/flutter
echo 'export PATH="$PATH:~/flutter/bin"' >> ~/.zshrc

# 5. 安装CocoaPods
sudo gem install cocoapods

# 6. 验证安装
flutter doctor
```

class OnlineIDEManager {

        const editor = monaco.editor.create(document.getElementById(containerId), {
            value: this.getTemplateCode(language),
            language: language,
            theme: 'vs-dark',
            automaticLayout: true,
            fontSize: 14,
            minimap: { enabled: true },
            scrollBeyondLastLine: false
        });
        
        this.editors.set(containerId, editor);
        this.setupAutoSave(editor, containerId);
        this.setupCollaboration(editor, containerId);
        
        return editor;
    }
    
    // 代码编译运行
    async compileAndRun(code, language) {
        try {
            const response = await fetch('/api/compile', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    code: code,
                    language: language,
                    timeout: 30000
                })
            });
            
            const result = await response.json();
            this.displayOutput(result);
            return result;
        } catch (error) {
            console.error('编译运行失败:', error);
            this.displayError(error.message);
        }
    }
    
    // 实时协作编辑
    setupCollaboration(editor, roomId) {
        const socket = io('/collaboration');
        
        socket.emit('join-room', roomId);
        
        editor.onDidChangeModelContent((e) => {
            socket.emit('content-change', {
                roomId: roomId,
                changes: e.changes,
                timestamp: Date.now()
            });
        });
        
        socket.on('content-update', (data) => {
            editor.executeEdits('remote-edit', data.changes);
        });
    }
    
    // 项目文件管理
    setupFileExplorer(projectId) {
        const fileExplorer = new FileExplorer({
            container: '#file-explorer',
            projectId: projectId,
            onFileSelect: (file) => {
                this.openFile(file);
            },
            onFileCreate: (name, type) => {
                this.createFile(name, type);
            },
            onFileDelete: (file) => {
                this.deleteFile(file);
            }
        });
        
        return fileExplorer;
    }

class VirtualTerminal {
    constructor(containerId) {
        this.terminal = new Terminal({
            fontSize: 14,
            fontFamily: 'Monaco, Menlo, "Ubuntu Mono", monospace',
            theme: {
                background: '#1e1e1e',
                foreground: '#ffffff'
            }
        });
        
        this.fitAddon = new FitAddon();
        this.terminal.loadAddon(this.fitAddon);
        
        this.terminal.open(document.getElementById(containerId));
        this.fitAddon.fit();
        
        this.setupCommandHandling();
    }
    
    setupCommandHandling() {
        this.terminal.onData((data) => {
            // 处理用户输入
            this.handleInput(data);
        });
    }
    
    async executeCommand(command) {
        try {
            const response = await fetch('/api/terminal', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    command: command,
                    sessionId: this.sessionId
                })
            });
            
            const result = await response.json();
            this.terminal.write(result.output);
        } catch (error) {
            this.terminal.write(`\r\n错误: ${error.message}\r\n`);
        }
    }
}
